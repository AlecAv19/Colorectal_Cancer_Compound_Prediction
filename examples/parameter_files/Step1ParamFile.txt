##########################
# Step 1. Parameter file
##########################

##########################
# Smiles Curation
#     Fix_Salts: If False, any molecule presented as a salt will be excluded. If True, the biggest fragment is selected and sanatized
#     Include_Isomers: If Ture, the final Smile will be writed in the isomeric form. This will be included in the descriptors computation.
#     Ignore_Charge: If True, charged molecules will be excluded
##########################
Fix_Salts = False
Include_Isomers = True
Ignore_Charge = True

##########################
# Molecule Descriptors
#     Descriptors_Type:
#     1: ECFP4 (1024)
#     2: ECFP4 (2048)
#     3: Rdkit Fingerprint
##########################
Descriptors_Type = 1

##########################
# Data Balancing by reduction
# The majority class is reduced to the size of the minority class using a clustering algorithms and randomly
# selecting from each cluster. The algorithm use the Kmean for clustering and the "Metric_Cluster" to compute the
# Silhouette score to define the optimal number of clusters.
#     Partition_Ratio: If the fraction Minoritary/Majoritary is lower than Partition_Ratio, the balancing is performed
#     N_Cluster_Min: Minimal number of clusters to consider
#     Metric_Cluster: braycurtis, canberra, chebyshev, correlation, dice, hamming, jaccard, kulsinski, mahalanobis,
#                     matching, minkowski, rogerstanimoto, russellrao, seuclidean, ‘sokalmichener’, sokalsneath,
#                     sqeuclidean, yule, cityblock, cosine, euclidean, l1, l2, manhattan.
#                     This metric is used to compute the Silhouette score
##########################
Partition_Ratio = 0.95
N_Cluster_Min = 5
Metric_Cluster = jaccard

##########################
# Data Partition
# The data will be split in several two possible forms. Normalization can be included as well as data reduction
#     Split_Ratio: 80-20 or 60-20-20. The number can be modified (sum is 100) but the data will be splited in two or three groups.
#                  if two the labels are training and external, if three the labels are training, test, external.
##########################
Split_Ratio = 60-20-20
